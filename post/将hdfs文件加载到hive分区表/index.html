#!/bin/bash
cu_date=`date +%Y%m%d`
begin_date="20181211"

while [ "$begin_date" -le "$cu_date" ];
do
	echo "${begin_date}"
	
	hive -e "alter table table_name add partition (dt='${begin_date}') 
	locatition 'hdfs://Cluster/user/hive/warehouse/.../dt=${begin_date}'; "
	
	begin_date=$(date -d "${begin_date}+1days" +%Y%m%d)
done
echo "finished"
